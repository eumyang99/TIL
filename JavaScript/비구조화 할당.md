### 배열 비구조화 할당

```javascript
let arr = ["one", "two", "three"]
let [one, two, three] = arr

/*
결과
one = "one"
two = "two"
three = "three"
와 같음
*/

let [one, two, three] = ["one", "two", "three"]

// 이것도 같은 코드
```



### 객체 비구조화 할당

```javascript
let object = {one:"one", two:"two", three:"three"}
let { three, two, one } = object

// 해당 key값의 value가 담김
// three = "three", two = "two", one="one" 

// 하지만 변수명이 key값으로 강제됨
let { three: mythree, two: mytwo, one: myone } = object
// object의 three라는 key값의 value를 mythree라는 변수명에 담음
// mythree = "three", mytwo = "two", myone="one" 
```
